File Uploader

Step 1: Add the JS files after the wrapper div. 


<!-- Drag-Drop-Uploader -->

<script src="vendor/jqueryui/jquery-ui.min.js"></script>

<script src="vendor/plugins/form/drag-drop-uploader/jquery.knob.js"></script>

<script src="vendor/plugins/form/drag-drop-uploader/script.js"></script>

<script src="vendor/plugins/form/drag-drop-uploader/jquery.iframe-transport.js"></script>

<script src="vendor/plugins/form/drag-drop-uploader/jquery.fileupload.js"></script>

Step 2: Add the HTML. 


<form id="upload" method="post" action="vendor/plugins/form/drag-drop-uploader/upload.php" enctype="multipart/form-data">

    

    <div id="drop" class="drop-uploader">

        <span class="drop-text hidden-xs">

            Drag &amp; Drop Files

            <br>

            <small>or</small>

        </span>

        

        <a class="btn btn-primary">

            Browse Files

        </a>

        

        <input type="file" name="upl" multiple />

        

        <ul>

            <!-- The file uploads will be shown here -->

        </ul>

    </div><!-- drop -->

    

</form>

Color Pickers

Quarca uses jQuery MiniColors 2.1 for color pickers.

Step 1: Add the CSS inside of the head tag. 


<link href="vendor/plugins/form/minicolors/minicolors.css" rel="stylesheet">

Step 2: Add the plugin's JS after the wrapper div. 


<script src="vendor/plugins/form/minicolors/jquery.minicolors.min.js"></script>

Step 3: Add the HTML & initialize the plugin.

$('.color-picker').each( function() {

    $(this).minicolors({

        control: $(this).attr('data-control') || 'hue',

        defaultValue: $(this).attr('data-defaultValue') || '',

        inline: $(this).attr('data-inline') === 'true',

        letterCase: $(this).attr('data-letterCase') || 'lowercase',

        opacity: $(this).attr('data-opacity'),

        position: $(this).attr('data-position') || 'bottom left',

        change: function(hex, opacity) {

            if( !hex ) return;

            if( opacity ) hex += ', ' + opacity;

            if( typeof console === 'object' ) {

                console.log(hex);

            }

        },

        theme: 'bootstrap'

    });

});

Date & Time Picker

Quarca uses Bootstrap 3 Date/Time Picker for date/time pickers.

Step 1: Add the CSS inside of the head tag. 


<link href="vendor/plugins/form/datetimepicker/bootstrap-datetimepicker.css" rel="stylesheet">

Step 2: Add the plugin's JS after the wrapper div. 


<script src="vendor/plugins/form/datetimepicker/bootstrap-datetimepicker.min.js"></script>

Step 3: Add the HTML & initialize the plugin.

  • Preview
  • HTML
  • JS

<!-- Date -->

<div class="form-group">

    <label>Date</label>

    <div class="input-group date" id="date">

	    <span class="input-group-addon"><i class="fa fa-calendar"></i></span>

	    <input type="text" class="form-control" placeholder="mm/dd/yyyy">

    </div>

</div>

<!-- Time -->

<div class="form-group">

    <label>Time</label>

    <div class="input-group date" id="time">

	    <span class="input-group-addon"><i class="fa fa-clock-o"></i></span>

	    <input type="text" class="form-control" placeholder="h:m AM/PM">

    </div>

</div>

<!-- Date & Time -->

<div class="form-group">

    <label>Date & Time</label>

    <div class="input-group date" id="date-time">

	    <span class="input-group-addon"><i class="fa fa-calendar"></i></span>

	    <input type="text" class="form-control" placeholder="mm/dd/yyyy h:m AM/PM">

    </div>

</div>
