Input Masks

Quarca uses this Masked Input plugin to enable specific data formatting.
Step 1: Add the plugin JS after the wrapper div.
<script src="vendor/plugins/form/masked-input/jquery.maskedinput.min.js"></script>
Step 2: Add the HTML & initialize the plugin.

* Check out the website for more options.

  • Preview
  • HTML
  • JS

<!-- Date -->

<div class="form-group">

    <label>Date</label>

    <input type="text" class="form-control" id="date" placeholder="mm/dd/yyyy">

</div>

<!-- Phone Number -->

<div class="form-group">

    <label>Phone Number</label>

    <input type="text" class="form-control" id="phone" placeholder="(999) 999-9999">

</div>

<!-- Tax ID -->

<div class="form-group">

    <label>Tax ID</label>

    <input type="text" class="form-control" id="tin" placeholder="99-9999999">

</div>

<!-- SSN -->

<div class="form-group last">

    <label>SSN</label>

    <input type="text" class="form-control" id="ssn" placeholder="999-99-9999">

</div>



jQuery(function($){

    //Date

    $("#date").mask("99/99/9999",{placeholder:"mm/dd/yyyy"});

    

    //Phone Number

    $("#phone").mask("(999) 999-9999");

    

    //Tax ID

    $("#tin").mask("99-9999999");

    

    //SSN

    $("#ssn").mask("999-99-9999");

});
